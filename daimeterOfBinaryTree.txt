        self.val=val
        self.left=left

def daimeterOfBinaryTree(root):
    daimeter=0
    def hieght(node):
        nonlocal daimeter
        if not node:
            return 0
        lh=hieght(node.left)
        rh=hieght(node.right)
        daimeter=max(daimeter,lh+rh)
        return 1+max(lh,rh)
    hieght(root)
    return daimeter
